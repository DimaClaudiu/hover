Define a vectorizer and use it to cast the dataset to 2D: 
import spacy
import re

nlp = spacy.load("en_core_web_md")

def vectorizer(text):
    clean_text = re.sub(r"[\s]+", r" ", text)
    return nlp(clean_text, disable=nlp.pipe_names).vector

vec = vectorizer("Winter is coming.")
print(f"Shape: {vec.shape}")
print(f"Vector: {vec}")
